SELECT FirstName, LastName, SUM(Defensive_Rebounds)
FROM
(
	SELECT PlayerID, FirstName, LastName, GameID, Date, Defensive_Rebounds
	FROM player_stats_in_game
	JOIN
	(
    		SELECT GameID, Date, Defensive_Rebounds
		FROM game
		WHERE	Date BETWEEN ‘(?)-08-01' AND ‘(?+1)-06-31'
	) as gameRange USING(GameID)
	JOIN 
    	(
        	SELECT PlayerID, FirstName, LastName
    		FROM player
    		WHERE	LastName = ‘?’
		AND	FirstName = ‘?’
    	) as playerSelect USING(PlayerID)
) as totalDR
